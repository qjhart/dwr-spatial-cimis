#+name: date_by_var
#+begin_src sql :tangle no :var v="day_air_tmp_max"
  with s as (
    select station_id,date,$v as s_v,$v_qc as s_qc
      from station where ref='s' and $v is not null and $v_qc not in ('M','S')
  ),
    d as (
      select station_id,date,$v as d_v,$v_qc as d_qc
        from station where ref='d'  and $v is not null and $v_qc not in ('M','S')
    ),
    j as (
      select * from s full outer join d using (station_id,date)
    ),
  md as (
  select date,count(*) as d_null from j where d_v is null group by date
  ),
  ms as (
  select date,count(*) as s_null from j where s_v is null group by date
  ),
  df as (
  select date,count(*) as diff from j where d_v != s_v group by date
  )
  select '$v' as var,* from ms full outer join md using (date) full outer join df using (date) order by date

#+end_src
